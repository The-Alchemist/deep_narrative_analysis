34def display_metadata(narrative_name: str, narrator: str, store_name: str): 

35 """ 

36 Display the metadata for a narrator (such as gender and birth year) and the narrative text. 

37 

38 @param narrative_name: The narrative title 

39 @param narrator: The URI with the narrator's metadata 

40 @param store_name: The database/store to be queried for data 

41 :return None 

42 """ 

43 logging.info(f'Displaying metadata and text for {narrative_name}') 

44 narrator_names = [] 

45 metadata_dict = dict() 

46 try: 

47 narrative_text_results = query_database( 

48 'select', query_narrative_text.replace('narrative_name', narrative_name), store_name) 

49 if narrative_text_results: 

50 narrative_text = narrative_text_results[0]['text']['value'] 

51 else: 

52 narrative_text = 'Error retrieving the narrative text. It cannot be displayed.' 

53 metadata1_results = query_database( 

54 'select', query_metadata1.replace("?narrator", f':{narrator}'), store_name) 

55 for binding in metadata1_results: 

56 narrator_names.append(binding['name']['value']) 

57 metadata2_results = query_database( 

58 'select', query_metadata2.replace("?narrator", f':{narrator}'), store_name) 

59 if metadata2_results: 

60 for binding in metadata2_results: 

61 # There should only be one result / one set of metadata for the narrator 

62 metadata_dict['country'] = binding['country']['value'] if 'country' in binding.keys() else 'Unknown' 

63 metadata_dict['year'] = binding['year']['value'] if 'year' in binding.keys() else 'Unknown' 

64 else: 

65 metadata_dict['country'] = 'Unknown' 

66 metadata_dict['year'] = 'Unknown' 

67 metadata3_results = query_database( 

68 'select', query_metadata3.replace("?narrator", f':{narrator}'), store_name) 

69 if metadata3_results: 

70 gender = empty_string 

71 for binding in metadata3_results: 

72 aspect = binding['aspect']['value'].split(':')[-1] 

73 if aspect in ('Agender', 'Bigender', 'Female', 'Male'): 

74 gender = aspect 

75 if gender: 

76 metadata_dict['gender'] = gender 

77 else: 

78 metadata_dict['gender'] = 'Unknown' 

79 else: 

80 metadata_dict['gender'] = 'Unknown' 

81 if not (metadata1_results or metadata2_results or metadata3_results): 

82 sg.popup_error(f'Limited or no metadata was found for the narrator, {narrator.split(":")[-1]}. ' 

83 f'The narrative text will be displayed, if available.', 

84 font=('Arial', 14), button_color='dark blue', icon=encoded_logo) 

85 except Exception as e: 

86 capture_error(f'Exception getting narrator details from {store_name}: {str(e)}', True) 

87 return 

88 

89 # Setup the PySimpleGUI window 

90 sg.theme('Material2') 

91 layout = [[sg.Text("Narrative Title:", font=('Arial', 16)), 

92 sg.Text(narrative_name, font=('Arial', 16))], 

93 [sg.Text()], 

94 [sg.Text("Narrator:", font=('Arial', 16)), 

95 sg.Text(', '.join(narrator_names), font=('Arial', 16))], 

96 [sg.Text()], 

97 [sg.Text("Narrator Gender:", font=('Arial', 16)), 

98 sg.Text(metadata_dict['gender'], font=('Arial', 16))], 

99 [sg.Text("Narrator Birth Country:", font=('Arial', 16)), 

100 sg.Text(metadata_dict['country'], font=('Arial', 16))], 

101 [sg.Text("Narrator Birth Year:", font=('Arial', 16)), 

102 sg.Text(metadata_dict['year'], font=('Arial', 16))], 

103 [sg.Text()], 

104 [sg.Text("Text:", font=('Arial', 16))], 

105 [sg.Multiline(key='narr_text', font=('Arial', 14), size=(75, 30), auto_refresh=True, 

106 autoscroll=True, background_color='#fafafa', text_color='black', write_only=True)], 

107 [sg.Text()], 

108 [sg.Text("To exit, close the window.", font=('Arial', 16))]] 

109 window_metadata_list = sg.Window(f'Metadata for {narrative_name}', layout, icon=encoded_logo).Finalize() 

110 window_metadata_list.find_element('narr_text', True).Update(narrative_text) 

111 window_metadata_list.find_element('narr_text', True).Widget.configure() 

112 # Non-blocking window 

113 window_metadata_list.read(timeout=0) 

114 return

117def display_timeline(narrative_name: str, event_list: list, store_name: str): 

118 """ 

119 Displays a timeline of the narrative or domain events using a matplotlib stem plot. 

120 

121 @param narrative_name: The name/label of the narrative 

122 @param event_list: An array of the binding results for the query, query_timeline_events, 

123 for the specified narrative 

124 @param store_name: The database/store to be queried for data 

125 @return: None (timeline is displayed) 

126 """ 

127 logging.info(f'Displaying narrative timeline for {narrative_name}') 

128 event_dict = dict() 

129 for binding in event_list: 

130 if 'month' in binding.keys(): 

131 dict_key = f'{binding["year"]["value"]}-{binding["month"]["value"]}' 

132 else: 

133 dict_key = f'{binding["year"]["value"]}-01' 

134 add_to_dictionary_values(event_dict, dict_key, binding['label']['value'], str) 

135 

136 dates = [] 

137 texts = [] 

138 for key, value in event_dict.items(): 

139 dates.append(key) 

140 texts.append('\n'.join(value)) 

141 # For matplotlib timeline, need datetime formatting 

142 plot_dates = [datetime.strptime(d, "%Y-%m") for d in dates] 

143 # Create a stem plot with some variation in levels as to distinguish close-by events. 

144 # Add markers on the baseline with dates 

145 # For each event, add a text label via annotate, which is offset from the tip of the event line 

146 levels = np.tile([-10, 10, -6, 6, -2, 2, -8, 8, -4, 4, -1, 1, -9, 9, -5, 5], 

147 int(np.ceil(len(plot_dates) / 6)))[:len(plot_dates)] 

148 # Create figure and plot a stem plot with the dates 

149 fig, ax = plt.subplots(figsize=(12, 8)) 

150 dpi = fig.get_dpi() 

151 fig.set_size_inches(808 * 2 / float(dpi), 808 / float(dpi)) 

152 ax.set(title=narrative_name) 

153 ax.vlines(plot_dates, 0, levels, color="tab:red") # The vertical stems 

154 ax.plot(plot_dates, np.zeros_like(plot_dates), "-o", 

155 color="k", markerfacecolor="w") # Baseline and markers on it 

156 # Annotate lines 

157 for d, l, r in zip(plot_dates, levels, texts): 

158 ax.annotate(r, xy=(d, l), 

159 xytext=(-2, np.sign(l) * 3), textcoords="offset points", 

160 horizontalalignment="right", 

161 verticalalignment="bottom" if l > 0 else "top", fontsize='x-small') 

162 # Format x-axis with yearly intervals 

163 ax.xaxis.set_major_locator(mdates.YearLocator()) 

164 ax.xaxis.set_major_formatter(mdates.DateFormatter("%Y-%m")) 

165 plt.setp(ax.get_xticklabels(), rotation=30, ha="right", fontsize='small') 

166 # Remove y axis and spines 

167 ax.yaxis.set_visible(False) 

168 ax.spines[["left", "top", "right"]].set_visible(False) 

169 ax.margins(y=0.1) 

170 plt.tight_layout(pad=0.05, h_pad=0.05, w_pad=0.05) 

171 plt.show() 

172 

173 # Display in a window with matplotlib interactive controls 

174 # TODO: Add checkboxes to limit relationships that are displayed? 

175 layout = [[sg.Text('Display Event Network for Date (YYYY-mm): ', font=('Arial', 14)), 

176 sg.InputText(text_color='black', background_color='#ede8e8', size=(10, 1), 

177 font=('Arial', 14), key='event_date', do_not_clear=True), 

178 sg.Button('Graph', button_color='dark blue', font=('Arial', 14), size=(6, 1))], 

179 [sg.Text('Controls:', font=('Arial', 14)), 

180 sg.Canvas(key='controls_cv')], 

181 [sg.Column(layout=[[sg.Canvas(key='fig_cv', size=(800 * 2, 800))]], pad=(0, 0))]] 

182 window_timeline = sg.Window('Timeline', layout, icon=encoded_logo, element_justification='center', 

183 resizable=True).Finalize() 

184 window_timeline['event_date'].Widget.config(insertbackground='black') 

185 draw_figure_with_toolbar(window_timeline['fig_cv'].TKCanvas, fig, 

186 window_timeline['controls_cv'].TKCanvas) 

187 window_timeline.Maximize() 

188 while True: 

189 event, values = window_timeline.read() 

190 if event == 'Graph': 

191 display_graph(narrative_name, event_list, store_name, values['event_date']) 

192 if event == sg.WIN_CLOSED: 

193 break 

194 window_timeline.close() 

195 return
