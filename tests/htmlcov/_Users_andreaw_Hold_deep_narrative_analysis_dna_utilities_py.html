65def add_to_dictionary_values(dictionary: dict, key: str, value, value_type): 

66 """ 

67 Add the value (cast using the value_type parameter) for the specified key in the dictionary 

68 if that key exists, or create a new dictionary entry for the key (where the value is a list). 

69 

70 :param dictionary: The dictionary to be updated 

71 :param key: String holding the dictionary key 

72 :param value: The value to be added to the current list of values of the dictionary entry, or 

73 a new list is created with the value 

74 :param value_type: The type of the value 

75 :return: None (Dictionary is updated) 

76 """ 

77 values = [] 

78 if key in dictionary.keys(): 

79 values = dictionary[key] 

80 values.append(value_type(value)) 

81 dictionary[key] = values 

82 return
