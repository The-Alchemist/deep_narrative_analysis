33def display_graph(narrative_name: str, event_list: list, store_name: str, event_date: str): 

34 """ 

35 Displays a graph of the narrative events using NetworkX. 

36 

37 @param narrative_name: The name/label of the narrative 

38 @param event_list: An array of the binding results for the query, query_timeline_events, for the 

39 specified narrative 

40 @param store_name: The database/store that holds the narrative's knowledge graph 

41 @param event_date: A string in YYYY-mm format indicating which events (events with times in that 

42 month) should be shown in the graph 

43 @return: None (graph is displayed) 

44 """ 

45 logging.info(f'Displaying narrative graph for {narrative_name}') 

46 # Get the details (predicates and objects) for each event individual 

47 node_set = set() 

48 edges = [] 

49 edge_label_dict = dict() 

50 for binding in event_list: 

51 event_uri = binding['event']['value'] 

52 event_name = event_uri.split(':')[-1] 

53 event_time = binding['year']['value'] 

54 if 'month' in binding.keys(): 

55 month = binding['month']['value'] 

56 if len(month) == 1: 

57 month = f'0{month}' 

58 event_time = f'{event_time}-{month}' 

59 else: 

60 event_time = f'{event_time}-01' 

61 if event_time != event_date: 

62 continue 

63 sentiment = get_sentiment(event_uri, narrative_name, store_name) 

64 event_tuples = get_event_data(narrative_name, store_name, event_uri) 

65 for event_tuple in event_tuples: 

66 if len(event_tuple) == 4: 

67 pred, obj, obj_label, obj_type = event_tuple 

68 else: 

69 pred, obj = event_tuple 

70 if obj.startswith('urn:'): 

71 obj_type = "Class" 

72 else: 

73 obj_type = empty_string 

74 obj_label = obj 

75 obj_node_name = obj_label 

76 if obj_type: 

77 obj_label = obj.split(':')[-1] 

78 if obj_type == 'Event': 

79 obj_sentiment = get_sentiment(obj, narrative_name, store_name) 

80 # Capture the sentiment with the node label so that pos/neg colors can be shown 

81 obj_node_name = f'evt{obj_label}$${str(obj_sentiment)}' 

82 elif obj_type == 'Class': 

83 obj_node_name = f'cls{obj_label}' 

84 else: 

85 obj_node_name = f'uri{obj_label}' 

86 if '#' in pred: 

87 pred_name = pred.split('#')[-1] 

88 else: 

89 pred_name = pred.split(':')[-1] 

90 edges.append((event_name, obj_label)) 

91 edge_label_dict[(event_name, obj_label)] = pred_name 

92 node_set.add(f'evt{event_name}$${str(sentiment)}') # Capture the sentiment with the node name 

93 node_set.add(obj_node_name) 

94 

95 # Associate the color red with 'bad' Events/States and green with 'good' ones 

96 # Associate the color purple with class types, blue with other URIs, and orange with strings/literals 

97 nodes = [] 

98 node_colors = [] 

99 node_sizes = [] 

100 for node in node_set: 

101 if node.startswith('evt'): 

102 node_details = node[3:].split('$$') 

103 nodes.append(node_details[0]) 

104 if float(node_details[1]) >= 0.0: 

105 node_colors.append('green') 

106 else: 

107 node_colors.append('red') 

108 node_sizes.append(750) 

109 elif node.startswith('cls'): 

110 nodes.append(node[3:]) 

111 node_colors.append('purple') 

112 node_sizes.append(200) 

113 elif node.startswith('uri'): 

114 nodes.append(node[3:]) 

115 node_colors.append('blue') 

116 node_sizes.append(350) 

117 else: 

118 nodes.append(node) 

119 node_colors.append('orange') 

120 node_sizes.append(100) 

121 

122 fig, ax = plt.subplots(figsize=(12, 8)) 

123 dpi = fig.get_dpi() 

124 fig.set_size_inches(808 * 2 / float(dpi), 808 / float(dpi)) 

125 dig = nx.DiGraph() 

126 dig.add_nodes_from(nodes) 

127 dig.add_edges_from(edges) 

128 pos = nx.spiral_layout(dig, scale=2) 

129 nx.draw(dig, pos, edge_color='black', width=1, linewidths=1, node_size=node_sizes, 

130 node_color=node_colors, alpha=0.5, with_labels=True) 

131 nx.draw_networkx_edge_labels(dig, pos, edge_labels=edge_label_dict, font_color='red') 

132 plt.tight_layout(pad=0.05, h_pad=0.05, w_pad=0.05) 

133 plt.show() 

134 

135 # Display in a window with matplotlib interactive controls 

136 layout = [[sg.Text('Controls:', font=('Arial', 14)), 

137 sg.Canvas(key='controls_cv')], 

138 [sg.Text('Legend:', font=('Arial', 14))], 

139 [sg.Text('Red', font=('Arial', 14), text_color='red'), 

140 sg.Text(' - Negative Events/States, ', font=('Arial', 14)), 

141 sg.Text('Green', font=('Arial', 14), text_color='green'), 

142 sg.Text('- Positive Events/States,', font=('Arial', 14))], 

143 [sg.Text('Purple', font=('Arial', 14), text_color='purple'), 

144 sg.Text(' - Class Types, ', font=('Arial', 14)), 

145 sg.Text('Blue', font=('Arial', 14), text_color='blue'), 

146 sg.Text('- Agents/Locations/Times/Resources, ', font=('Arial', 14)), 

147 sg.Text('Orange', font=('Arial', 14), text_color='orange'), 

148 sg.Text('- Literals', font=('Arial', 14))], 

149 [sg.Column(layout=[[sg.Canvas(key='fig_cv', size=(800 * 2, 800))]], pad=(0, 0))]] 

150 window_graph = sg.Window('Graph', layout, icon=encoded_logo, element_justification='center', 

151 resizable=True).Finalize() 

152 draw_figure_with_toolbar(window_graph['fig_cv'].TKCanvas, fig, 

153 window_graph['controls_cv'].TKCanvas) 

154 window_graph.Maximize() 

155 while True: 

156 event, values = window_graph.read(timeout=0) 

157 if event == sg.WIN_CLOSED: 

158 break 

159 window_graph.close() 

160 return

163def get_event_data(narrative_name: str, store_name: str, event_uri: str) -> list: 

164 """ 

165 Get details (predicates and objects) for an event. 

166 

167 @param narrative_name: String holding the narrative title/label 

168 @param store_name: String holding the database/store name from which the event data is retrieved 

169 @param event_uri: The URI of the event in the database/data store 

170 @return: An array of tuples of (predicate value, object uri, object value, object type) for the 

171 related entities and properties of the event 

172 """ 

173 results = [] 

174 try: 

175 query_str = query_graph_individuals.replace('uri', event_uri).\ 

176 replace('narr_graph', narrative_name.replace(' ', '_')) 

177 event_details = query_database('select', query_str, store_name) 

178 if not event_details: 

179 capture_error(f'No event triples were returned for the uri, {event_uri}, in ' 

180 f'the narrative, {narrative_name}, in the database, {store_name}.', True) 

181 return results 

182 except Exception as e: 

183 capture_error( 

184 f'Exception getting event details for {event_uri} for the narrative, {narrative_name}, ' 

185 f'in the database, {store_name}: {str(e)}', True) 

186 return [] 

187 for binding in event_details: 

188 obj_type = empty_string 

189 if 'e_type' in binding: 

190 obj_type = "Event" 

191 elif 'a_type' in binding: 

192 obj_type = "Agent" 

193 elif 't_type' in binding: 

194 obj_type = "Time" 

195 elif 'r_type' in binding: 

196 obj_type = "Resource" 

197 if obj_type: 

198 results.append((binding['pred']['value'], binding['obj']['value'], 

199 binding['obj_label']['value'], obj_type)) 

200 else: 

201 results.append((binding['pred']['value'], binding['obj']['value'])) 

202 return results

205def get_sentiment(event_uri: str, narrative_name: str, store_name: str) -> float: 

206 """ 

207 Get the value of the :sentiment predicate for the event. 

208 

209 @param event_uri: String holding the URI to be queries 

210 @param narrative_name: String holding the narrative where the event is defined 

211 @param store_name: String holding the database where the narrative is stored 

212 :return Float that is the event's sentiment 

213 """ 

214 sentiment_details = query_database( 

215 'select', 

216 query_event_sentiment.replace('uri', event_uri).replace('narr_graph', narrative_name.replace(' ', '_')), 

217 store_name) 

218 if sentiment_details: 

219 return sentiment_details[0]['sentiment']['value'] 

220 else: 

221 return 0.0 # Default sentiment is neutral
